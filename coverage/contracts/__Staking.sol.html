// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;
 
import "@openzeppelin/contracts/access/Ownable.sol";
import "@openzeppelin/contracts/token/ERC20/IERC20.sol";
import "@openzeppelin/contracts/utils/math/SafeMath.sol";
 
contract __Staking is Ownable {
    using SafeMath for uint256;
 
    mapping(address => Stakeholder) internal stakeholders;
    IERC20 public token;
    uint256 public rewardRate;
    uint256 public totalStaked;
 
    struct Stakeholder {
        uint256 since;
        uint256 amount;
        uint256 userStakingTime;
        bool active;
    }
 
    // event NewStake (
    //   address from,
    //   uint timeOfStake,
    //   uint amount,
    //   uint userStakingTime // set to zero after unstaked
    // );
 
    constructor(address _token, uint256 _rewardRate) {
        token = IERC20(_token);
        rewardRate = _rewardRate;
    }
 
    function isStakeholder(address _address) public view returns (bool) {
        return stakeholders[_address].active;
    }
 
    function addStakeholder(address _stakeholder, uint256 _time) public {
        if (!isStakeholder(_stakeholder)) {
            Stakeholder storage stakeholder = stakeholders[_stakeholder];
            stakeholder.since = block.timestamp;
            stakeholder.amount = 0;
            stakeholder.userStakingTime = _time;
            stakeholder.active = true;
        }
    }
 
    function removeStakeholder(address _stakeholder) public {
        if (isStakeholder(_stakeholder)) {
            stakeholders[_stakeholder].active = false;
        }
    }
 
    function stakeOf(address _stakeholder) public view returns (uint256) {
        return stakeholders[_stakeholder].amount;
    }
 
    function totalStakes() public view returns (uint256) {
        return totalStaked;
    }
 
    function createStake(uint256 _stake, uint256 _time) public {
        if (!stakeholders[msg.sender].active) addStakeholder(msg.sender, _time);
 
        stakeholders[msg.sender].amount.add(_stake);
        stakeholders[msg.sender].since = block.timestamp;
 
        totalStaked = totalStaked.add(_stake);
 
        token.transferFrom(msg.sender, address(this), _stake);
    }
 
    function removeStake(uint256 _stake) public {
        require(
            block.timestamp - stakeholders[msg.sender].since >=
                stakeholders[msg.sender].userStakingTime
        );
        totalStaked = totalStaked.sub(_stake);
 
        if (stakeholders[msg.sender].amount == _stake)
            removeStakeholder(msg.sender);
 
        uint256 reward = calculateReward(msg.sender, _stake);
        uint256 totalReturn = _stake.add(reward);
 
        stakeholders[msg.sender].amount = stakeholders[msg.sender].amount.sub(
            _stake
        );
 
        token.transfer(msg.sender, totalReturn);
    }
 
    function calculateReward(address _stakeholder, uint256 _stake)
        public
        view
        returns (uint256)
    {
        Stakeholder storage stakeholder = stakeholders[_stakeholder];
        uint256 stakingTime = stakeholder.userStakingTime;
 
        uint256 rewardAmount = _stake
            .mul(rewardRate)
            .mul(stakingTime)
            .div(365 days)
            .div(100);
 
        return rewardAmount;
    }
}
